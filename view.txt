#creates view called 'joinedtables' of all four tables joined together #

create view joinedtables
as select
	 sampleID, age, gender, disease_state, source, probeID, expression,
	 experiment, probe_names, geneID, geneid, gene_symbol, gene_title
from
	 sample_info, gene_expression, probe, gene
where

sample_info inner join gene_expression on sample_info.sampleID = gene_expression.experiment
inner join probe onprobe.probe_names = gene_expresison.probeID
inner join gene on gene.geneid = probe.geneID;
